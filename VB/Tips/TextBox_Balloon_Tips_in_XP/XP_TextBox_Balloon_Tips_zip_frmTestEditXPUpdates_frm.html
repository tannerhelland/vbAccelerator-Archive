Attribute VB_Name = "frmTestEditXPUpdates"
Attribute VB_GlobalNameSpace = False
Attribute VB_Creatable = False
Attribute VB_PredeclaredId = True
Attribute VB_Exposed = False
Option Explicit

Implements ISubclass
Private Const WM_COMMAND = &H111
Private Const EN_MAXTEXT = &H501

Private m_cTextBoxXp As cTextBoxXP

Private Sub cmdShowInfoBalloon_Click()
   m_cTextBoxXp.BalloonTipText = "Enter any word to search for in the box.  If
    you want to find an exact phrase, enter the phrase in quotes.  To find only
    documents containing all specified words, write AND between the words."
   m_cTextBoxXp.BalloonTipIcon = TTI_INFO
   m_cTextBoxXp.BalloonTipTitle = "Using the Search Box"
   m_cTextBoxXp.ShowBalloonTip
End Sub

Private Sub Command2_Click()
   m_cTextBoxXp.CueBanner = "Why doesn't this work?" & vbNullChar
End Sub

Private Sub Form_Load()
   Set m_cTextBoxXp = New cTextBoxXP
   m_cTextBoxXp.TextBox = txtSearch
   m_cTextBoxXp.CueBanner = "Enter your search here"
   AttachMessage Me, txtSearch.Container.hWnd, WM_COMMAND
End Sub

Private Property Let ISubclass_MsgResponse(ByVal RHS As SSubTimer.EMsgResponse)
   '
   '
End Property

Private Property Get ISubclass_MsgResponse() As SSubTimer.EMsgResponse
   '
   ISubclass_MsgResponse = emrPreprocess
   '
End Property

Private Function ISubclass_WindowProc(ByVal hWnd As Long, ByVal iMsg As Long,
 ByVal wParam As Long, ByVal lParam As Long) As Long
   '
   If (iMsg = WM_COMMAND) Then
      If (lParam = txtSearch.hWnd) Then
         Dim wNotify As Long
         wNotify = (wParam And &H7FFF0000) \ &H10000
         Debug.Print "Got WM_COMMAND", wNotify
         If (wNotify = EN_MAXTEXT) Then
            m_cTextBoxXp.BalloonTipIcon = TTI_WARNING
            m_cTextBoxXp.BalloonTipTitle = "Too many characters"
            m_cTextBoxXp.BalloonTipText = "You have entered too many characters
             into the search box: the search is limited to 20 characters."
            m_cTextBoxXp.ShowBalloonTip
         End If
      End If
   End If
   '
End Function
