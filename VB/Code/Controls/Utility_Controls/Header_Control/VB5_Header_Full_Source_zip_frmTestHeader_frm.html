Private Sub cboColumn_Click()
Dim lCol As Long
Dim eAlign As EHdrTextAlign
Dim i As Long

   If (cboColumn.ListIndex > -1) Then
      lCol = cboColumn.ItemData(cboColumn.ListIndex)
      
      txtHeader.Text = hdrColumns.ColumnHeader(lCol)
      txtTag.Text = hdrColumns.ColumnTag(lCol)
      txtExtraData.Text = hdrColumns.ColumnExtraData(lCol)
      txtWidth.Text = hdrColumns.ColumnWidth(lCol)
      cboOrder.ListIndex = hdrColumns.ColumnIndex(lCol)
      cboIcon.ListIndex = hdrColumns.ColumnImage(lCol) + 1
      cboAlign.ListIndex = hdrColumns.ColumnTextAlign(lCol) And &H3&
      chkImage.Value = Abs(hdrColumns.ColumnImageOnRight(lCol))
      txtFilter.Text = hdrColumns.ColumnFilter(lCol)
      cboGroupOrder.Enabled = (hdrColumns.AllowGrouping)
      cboGroupOrder.Clear
      cboGroupOrder.AddItem "None"
      If (hdrColumns.AllowGrouping) Then
         For i = 1 To hdrColumns.ColumnGroupCount
            cboGroupOrder.AddItem i - 1
         Next i
         If (hdrColumns.ColumnIsGrouped(lCol)) Then
            cboGroupOrder.ListIndex = hdrColumns.ColumnGroupOrder(lCol) + 1
         Else
            cboGroupOrder.ListIndex = 0
         End If
      Else
         cboGroupOrder.ListIndex = 0
      End If
   End If
   
End Sub

Private Sub cboIcon_Click()
Dim lIcon As Long
   picIcon.Cls
   lIcon = cboIcon.ItemData(cboIcon.ListIndex)
   If (lIcon > -1) Then
      ilsIcons.ListImages(lIcon + 1).Draw picIcon.hDC, 0, 0
      picIcon.Refresh
   End If
End Sub

Private Sub cmdDelete_Click()
   If (cboColumn.ListIndex < 0) Then
      MsgBox "Please choose a column", vbExclamation
   Else
      If (vbYes = MsgBox("Are you sure you want to delete this column?",
       vbYesNo Or vbQuestion)) Then
         hdrColumns.RemoveColumn cboColumn.ItemData(cboColumn.ListIndex)
         BuildLists
      End If
   End If
End Sub

Private Sub cmdRefresh_Click()
   cboColumn_Click
End Sub

Private Sub cmdSet_Click()
Dim lCol As Long
   If (cboColumn.ListIndex > -1) Then
      lCol = cboColumn.ItemData(cboColumn.ListIndex)
      With hdrColumns
         .ColumnHeader(lCol) = txtHeader.Text
         .ColumnWidth(lCol) = CLng(txtWidth.Text)
         .ColumnImage(lCol) = cboIcon.ItemData(cboIcon.ListIndex)
         .ColumnTag(lCol) = txtTag.Text
         .ColumnExtraData(lCol) = CLng(txtExtraData.Text)
         .ColumnTextAlign(lCol) = cboAlign.ListIndex
         .ColumnIndex(lCol) = cboOrder.ItemData(cboOrder.ListIndex)
         .ColumnImageOnRight(lCol) = (chkImage.Value = Checked)
         .ColumnFilter(lCol) = txtFilter.Text
         If (.AllowGrouping) Then
            If (cboGroupOrder.ListIndex > 0) Then
               .ColumnGroupOrder(lCol) = cboGroupOrder.ListIndex - 1
            Else
               ' not grouped
               .ColumnIsGrouped(lCol) = False
            End If
         End If
      End With
      cboColumn_Click
   End If
End Sub

Private Sub Form_Load()
Dim i As Long
   
   cboIcon.AddItem "<none>"
   cboIcon.ItemData(cboIcon.NewIndex) = -1
   
   For i = 1 To ilsIcons.ListImages.Count
      cboIcon.AddItem i - 1
      cboIcon.ItemData(cboIcon.NewIndex) = i - 1
   Next i
   cboAlign.AddItem "Left"
   cboAlign.AddItem "Right"
   cboAlign.AddItem "Center"
   
   
   With hdrColumns
      .AutoHeight = True
      
      .ImageList = ilsIcons
      .AddColumn "Column1"
      .AddColumn "Column2", 128, HdrTextALignRight
      .AddColumn "Column3"
      .AddColumn "Column4"
      .AddColumn "Column5"
       
      BuildLists
      
      hdrColumns_ColumnClick 0
      
      .ColumnImageOnRight(1) = True
   End With
    
End Sub
Private Sub BuildLists()
Dim iCol As Long
   cboColumn.Clear
   cboOrder.Clear
   For iCol = 1 To hdrColumns.ColumnCount
      cboColumn.AddItem hdrColumns.ColumnHeader(iCol - 1)
      cboColumn.ItemData(cboColumn.NewIndex) = iCol - 1
      cboOrder.AddItem iCol
      cboOrder.ItemData(cboOrder.NewIndex) = iCol - 1
   Next iCol
   If (cboColumn.ListCount > 0) Then
      cboColumn.ListIndex = 0
   End If

End Sub

Private Sub hdrColumns_RightClick(X As Single, Y As Single)
Dim i As Long
Dim mX As Long, mY As Long
   mX = X \ Screen.TwipsPerPixelX
   mY = Y \ Screen.TwipsPerPixelY
   i = hdrColumns.ColumnHitTest(mX, mY)
   Status "RightClick:" & X & "," & Y & "," & _
      IIf(i > -1, " Column:" & i, "Not on a column")
End Sub
