Private m_cIL As cVBALSysImageList
Private m_bInterlock As Boolean

Private Sub LoadPath(ByVal sPath As String)
Dim hFiles As Long
Dim fd As WIN32_FIND_DATA
Dim sName As String
Dim f As Boolean
Dim bSkip As Boolean, bHaveFile As Boolean
Dim iC As Long

   If Not m_bInterlock Then
      chkLargeIcons.Enabled = False
      m_bInterlock = True
      Screen.MousePointer = vbArrowHourglass
      lstFiles.Clear
   
      ' Get files in directory:
      hFiles = FindFirstFile(sPath & "*.*", fd)
      f = Not (hFiles = INVALID_HANDLE_VALUE)
      
      Do While f
         iC = iC + 1
         ' Add to list:
         sName = ByteZToStr(fd.cFileName)
         bSkip = False
         If (fd.dwFileAttributes And FILE_ATTRIBUTE_DIRECTORY) =
          FILE_ATTRIBUTE_DIRECTORY Then
            If bHaveFile Then
               lstFiles.InsertItemAndData sName, 0, m_cIL.ItemIndex(sPath &
                sName, True), , , , fd.dwFileAttributes, , m_cIL.IconSizeX, ,
                eixVCentre
               bSkip = True
            End If
         Else
            bHaveFile = True
         End If
         If Not bSkip Then
            lstFiles.AddItemAndData sName, m_cIL.ItemIndex(sPath & sName,
             True), , , , fd.dwFileAttributes, , m_cIL.IconSizeX, , eixVCentre
         End If
         
         ' ...
         If iC Mod 50 = 0 Then
            DoEvents
         End If
         ' Keep looping until no more files
          f = FindNextFile(hFiles, fd)
      Loop
      
      ' Done:
      f = FindClose(hFiles)
      Screen.MousePointer = vbDefault
      m_bInterlock = False
      chkLargeIcons.Enabled = True
   End If
End Sub
Private Function ByteZToStr(ByRef b() As Byte) As String
Dim iPos As Long
Dim s As String
   s = StrConv(b, vbUnicode)
   iPos = InStr(s, vbNullChar)
   If iPos > 0 Then
      ByteZToStr = left$(s, iPos - 1)
   Else
      ByteZToStr = s
   End If
End Function
Private Sub SetPath(ByVal sPath As String)
   If Right$(sPath, 1) <> "\" Then sPath = sPath & "\"
   lblPath.Caption = sPath
   LoadPath sPath
End Sub

Private Sub chkLargeIcons_Click()
   If chkLargeIcons.Value = Checked Then
      If m_cIL.IconSizeX <> 32 Then
         m_cIL.IconSizeX = 32
         m_cIL.Create
         lstFiles.ImageList = m_cIL.hIml
         SetPath lblPath.Caption
      End If
   Else
      If m_cIL.IconSizeX <> 16 Then
         m_cIL.IconSizeX = 16
         m_cIL.Create
         lstFiles.ImageList = m_cIL.hIml
         SetPath lblPath.Caption
      End If
   End If
End Sub

Private Sub Form_Load()
   
   Set m_cIL = New cVBALSysImageList
   m_cIL.IconSizeX = 16
   m_cIL.IconSizeY = 16
   m_cIL.Create
      
   lstFiles.ImageList = m_cIL.hIml
      
   Me.Show
   Me.Refresh
         
   SetPath "C:\"
   
End Sub

Private Sub Form_Resize()
Dim lH As Long
   On Error Resume Next
   lH = Me.ScaleHeight - lstFiles.tOp - chkLargeIcons.Height - 4 *
    Screen.TwipsPerPixelY
   lstFiles.Move lstFiles.left, lstFiles.tOp, Me.ScaleWidth - lstFiles.left *
    2, lH
   chkLargeIcons.Move chkLargeIcons.left, lstFiles.tOp + lstFiles.Height + 2 *
    Screen.TwipsPerPixelY
End Sub

Private Sub lstFiles_DblClick()
Dim eType As EWin32FileAttributes
Dim sFile As String
Dim iPos As Long
   
   If lstFiles.ListIndex > -1 Then
      sFile = lblPath.Caption & lstFiles.List(lstFiles.ListIndex)
      eType = lstFiles.ItemData(lstFiles.ListIndex)
      If (eType And FILE_ATTRIBUTE_DIRECTORY) = FILE_ATTRIBUTE_DIRECTORY Then
         If Not m_bInterlock Then
            Select Case lstFiles.List(lstFiles.ListIndex)
            Case "."
               ' refresh
               SetPath lblPath.Caption
            Case ".."
               ' Find previous directory
               sFile = lblPath.Caption
               ' InstrRev in VB6
               iPos = Len(sFile) - 1
               Do
                  If Mid$(sFile, iPos, 1) = "\" Then
                     sFile = left$(sFile, iPos - 1)
                     Exit Do
                  End If
                  iPos = iPos - 1
               Loop
               SetPath sFile
            Case Else
               ' New directory
               SetPath sFile
            End Select
         End If
      Else
         If vbYes = MsgBox("Start '" & sFile & "?", vbYesNo Or vbQuestion) Then
            ShellEx sFile
         End If
      End If
   End If
End Sub
