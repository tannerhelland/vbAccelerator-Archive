Private Sub LogEvent(ByVal sMsg As String)
   lstEvents.AddItem sMsg
   lstEvents.ListIndex = lstEvents.NewIndex
End Sub

Private Sub Form_Load()
Dim tabX As cTab
   tabLeft.ImageList = mfrmMain.ilsIcons
   Set tabX = tabLeft.Tabs.Add("EVENTS", , "Events")
   tabX.Panel = picEvents
End Sub

Private Sub Form_Resize()
   On Error Resume Next
   If Not (StrComp(tabLeft.Tag, "CLOSE")) = 0 Then
      txtDocument.Left = tabLeft.Left + tabLeft.Width + 2 *
       Screen.TwipsPerPixelX
   End If
   txtDocument.Move txtDocument.Left, txtDocument.Top, Me.ScaleWidth -
    txtDocument.Left - 2 * Screen.TwipsPerPixelX, Me.ScaleHeight -
    txtDocument.Top - 2 * Screen.TwipsPerPixelY
End Sub

Private Sub picEvents_Resize()
   lstEvents.Move 2 * Screen.TwipsPerPixelX, 2 * Screen.TwipsPerPixelY,
    picEvents.ScaleWidth - 2 * Screen.TwipsPerPixelX, picEvents.ScaleHeight - 2
    * Screen.TwipsPerPixelY
End Sub

Private Sub tabLeft_Pinned()
   LogEvent "Pinned"
   Form_Resize
End Sub
