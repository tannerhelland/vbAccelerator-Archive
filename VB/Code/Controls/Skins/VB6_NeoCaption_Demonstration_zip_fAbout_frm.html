Option Explicit

Private m_cN As New cNeoCaption
Private m_fOwner As Object

Public Property Let Owner(ByRef f As Object)
   Set m_fOwner = f
End Property

Private Sub cmdOK_Click()
   Unload Me
End Sub

Private Sub Form_Load()
Dim ctl As Control
   If Not m_fOwner Is Nothing Then
      Dim cSkin As cSkinConfiguration
      Set cSkin = m_fOwner.SkinConfiguration
      If Not (cSkin Is Nothing) Then
         m_cN.Attach2 Me, cSkin
         Me.BackColor = m_cN.MenuBackgroundColor
         For Each ctl In Me.Controls
            If TypeOf ctl Is Label Then
               If Not ctl.Name = "lblInfo" Then
                  ctl.ForeColor = m_cN.ActiveMenuColor
               End If
            End If
         Next
         txtSpecialCopyright.BackColor = Me.BackColor
         txtSpecialCopyright.ForeColor = m_cN.ActiveMenuColor
         cmdOk.BackColor = Me.BackColor
      End If
   End If
   Set m_fOwner = Nothing
End Sub

Private Sub lblCopyright_Click()
   'ShellEx lblCopyright.tag
End Sub

   'ShellEx lblProduct.tag
End Sub

Private Sub lblURL_Click()
   'ShellEx lblUrl.tag
End Sub

End Sub
