static INT gs_nDebug = 1;


void SetDebugLevel( int nValue )
{
   gs_nDebug= nValue;
}



void DebugPrintf(const char* pzFormat, ...)
{
   if (gs_nDebug)
   {
      BOOL bLogFile=TRUE;

      char szBuffer[8192];
      va_list ap;

      // Clear buffer
      memset(szBuffer,0x00,sizeof(szBuffer));

      // This will make the vsprint code only to run uf gs.bLogfile is set if
       we are a release build
   #if !defined _DEBUG && !defined SPECIALDEBUG
      if (bLogFile) {
   #endif
         va_start(ap, pzFormat);
         _vsnprintf(szBuffer, sizeof(szBuffer), pzFormat, ap);
   #if !defined _DEBUG && !defined SPECIALDEBUG
      }
   #endif

      if (bLogFile) 
      {
         FILE* fp;
         time_t t;
         struct tm* psTM;

         time(&t);
         psTM = localtime(&t);

         fp = fopen(".\\CDRipDllLog.txt", "a");
         if (fp) {
            fprintf(fp, "%04X-%02d%02d%02d (%d): %s\n", GetCurrentProcessId(),
             psTM->tm_hour, psTM->tm_min, psTM->tm_sec,GetTickCount(),szBuffer);

            fclose(fp);
         }
      }

   #if defined _DEBUG || defined SPECIALDEBUG
      strcat(szBuffer, "\n\r");
   //    OutputDebugString("CDex: ");
      OutputDebugString(szBuffer);
   #endif
      // This will make the va_end code only to run uf gs.bLogfile is set if we
       are a release build
   #if !defined _DEBUG && !defined SPECIALDEBUG
      if (bLogFile)
   #endif
         va_end(ap);
   }
}
