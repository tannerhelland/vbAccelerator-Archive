VERSION 1.0 CLASS
BEGIN
  MultiUse = -1  'True
END
Attribute VB_Name = "Addresses"
Attribute VB_GlobalNameSpace = False
Attribute VB_Creatable = True
Attribute VB_PredeclaredId = False
Attribute VB_Exposed = False
Option Explicit

Private m_colAddresses As Collection

Private Sub Class_Initialize()

   Set m_colAddresses = New Collection

End Sub

Private Sub Class_Terminate()

   Set m_colAddresses = Nothing

End Sub

Public Property Get Count() As Long

   Count = m_colAddresses.Count

End Property

Public Property Get Item(ByVal Index As Variant) As Address
Attribute Item.VB_UserMemId = 0

   Set Item = m_colAddresses.Item(Index)

End Property

Public Function Add() As Address

   Dim oAdd As New Address

   m_colAddresses.Add oAdd
   Set Add = oAdd

End Function
