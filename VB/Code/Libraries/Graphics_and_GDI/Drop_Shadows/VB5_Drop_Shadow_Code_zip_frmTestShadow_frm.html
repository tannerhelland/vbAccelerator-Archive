Attribute VB_Name = "frmTestDropShadow"
Attribute VB_GlobalNameSpace = False
Attribute VB_Creatable = False
Attribute VB_PredeclaredId = True
Attribute VB_Exposed = False
Option Explicit

Private m_fRight As fDropShadow
Private m_fBottom As fDropShadow

Private m_fRightPic As fDropShadow
Private m_fBottomPic As fDropShadow

Private m_fRightTab As fDropShadow


Private Sub chkShadows_Click()
   
   If (chkShadows.value = vbChecked) Then
      
      Set m_fRight = New fDropShadow
      m_fRight.ShadowSize = 8
      m_fRight.ShadowType = RightShadow
      m_fRight.Shadow = Me
      m_fRight.Show , Me
      
      Set m_fBottom = New fDropShadow
      m_fBottom.ShadowSize = 8
      m_fBottom.ShadowType = BottomShadow
      m_fBottom.Shadow = Me
      m_fBottom.Show , Me
      
      Set m_fRightPic = New fDropShadow
      m_fRightPic.ShadowType = RightShadow
      m_fRightPic.Shadow = picImage
      m_fRightPic.Show , Me
      
      Set m_fBottomPic = New fDropShadow
      m_fBottomPic.ShadowType = BottomShadow
      m_fBottomPic.Shadow = picImage
      m_fBottomPic.Show , Me
      
      Set m_fRightTab = New fDropShadow
      m_fRightTab.ShadowType = RightShadow
      m_fRightTab.Shadow = picTab
      m_fRightTab.Show , Me
      
      On Error Resume Next ' we might not be visible yet
      Me.SetFocus
      
      
   Else
      Unload m_fRight
      Set m_fRight = Nothing
      Unload m_fBottom
      Set m_fBottom = Nothing
      Unload m_fRightPic
      Set m_fRightPic = Nothing
      Unload m_fBottomPic
      Set m_fBottomPic = Nothing
      Unload m_fRightTab
      Set m_fRightTab = Nothing
   End If
End Sub

Private Sub Form_Load()
   chkShadows_Click
End Sub

Private Sub Form_QueryUnload(Cancel As Integer, UnloadMode As Integer)
   ' clear up:
   If (chkShadows.value = vbChecked) Then
      chkShadows_Click
   End If
End Sub

Private Sub lblShadow_Click()
   chkShadows.value = IIf(chkShadows.value = vbChecked, vbUnchecked, vbChecked)
End Sub
