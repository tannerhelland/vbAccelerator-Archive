Public WithEvents c As cBrowseForFolder
Attribute c.VB_VarHelpID = -1

Private Sub cmdCustom_Click()
   Me.Enabled = False
   Dim fC As New frmCapture
   Dim c As New cCaptureBF
   With c
      With .Browse
         .hwndOwner = Me.hwnd
         .InitialDir = App.Path
         .FileSystemOnly = True
         .Title = ""
      End With
      .Show fC
   End With
   If Not fC.Cancelled Then
      MsgBox "Choose folder: " & fC.SelectedFolder, vbInformation
   End If
   'Me.Enabled = True
   'Me.SetFocus
End Sub

Private Sub cmdStandard_Click()
Dim s As String

   c.hwndOwner = Me.hwnd
   c.InitialDir = App.Path
   c.FileSystemOnly = True
   c.StatusText = True
   c.EditBox = True
   c.UseNewUI = True
   s = c.BrowseForFolder
   If Len(s) > 0 Then
      MsgBox s, vbInformation
   End If

End Sub

Private Sub Form_Load()
   Set c = New cBrowseForFolder
End Sub
